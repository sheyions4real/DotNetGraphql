// calling the graphQL API via client

// To Query to get platforms and command
         query{
	    platform
	    {
		id
		name
		commands{
			id
			howTo
			commandLine
		}
	     }
         }



// Query with filter using the where clause
           query{
	      command (where: {platformId:{eq:1}}){
		id
		howTo
		commandLine
		platform{
			id
			name
		}
	      }
           }

// Query using sorting
	     query{
	      platform (order: {name:DESC}){
		id
		name
		commands{
			id
			commandLine	
		}
	      }
	     }


// To Add a new Platform
     mutation{
	addPlatform(input: {
		name: "Mac OS"
	}){
		platform{
			id
			name
		}
	}
      }

// to add a new command
      mutation{
	  addCommand(input: {
		howTo:"how to check the version of jekins"
		commandLine:"jekins -version"
		platformId:4
	  })
	  {
		command{
			id
			howTo
			commandLine
			platform
			{
				id
				name
			}
		}
	  }
        }



// Use mutation to add and delete

subscription

  onPlatformAdded {
